}
class plan {
- titl: string

- eventd: dat
+ addtask(task: task): void
+ createschedul(): schedul
+ allocatemateri(): void
}

- nam: string

+ assignmateri(materi: materi): void
}

- starttim: tim
- endtim: tim
+ addtasktoschedul(task: task): void
}
class materi {
- nam: string
- quantiti: integ
- typ: string
}
